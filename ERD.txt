
category
---
category_id PK INT
category_name VARCHAR(25)

subcategory
---
subcategory_id PK INT
subcategory_name VARCHAR(25)


contacts
---
contact_id PK INT
first_name VARCHAR(25)
last_name VARCHAR(25)
email   VARCHAR(60)

campaign
---
cf_id PK INT
contact_id INT FK >- contacts.contact_id
company_name VARCHAR(35)
description VARCHAR(35)
goal FLOAT
pledged FLOAT
outcome VARCHAR(35)
backers_count INT
country CHAR(2)
currency CHAR(3)
launched_data DATE
end_date DATE
category_id INT FK >- category.category_id
subcategory_id INT FK >- subcategory.subcategory_id
